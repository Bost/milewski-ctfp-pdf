;; Composition is associative
(defn f [x] (str "f" x "f"))
(defn g [x] (str "g" x "g"))
(defn h [x] (str "h" x "h"))

;; TODO compute as a test
(let [x 1]
  (= ((comp h (comp g f)) x) ((comp (comp h g) f) x)))
                    

;; Identity
(defn f [x] identity)
(defn ida [a] a)
(defn idb [b] b)

(let [x 1]
  (= ((comp f ida) x) (f x) ((comp idb f) x)))
